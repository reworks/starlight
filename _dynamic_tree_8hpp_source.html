195  [[maybe_unused]] inline const bool update(const key_type& key, AABB aabb, bool force_reinsert = false)
196  {
197  if (const auto it = m_index_map.find(key); it != m_index_map.end())
198  {
199  const auto node_index = it->second; // Extract the node index.
200 
201  // No need to update if the particle is still within its fattened AABB.
202  if (!force_reinsert && m_nodes.at(node_index).aabb.contains(aabb))
203  {
204  return false;

207  // Remove the current leaf.
208  remove_leaf(node_index);
209  aabb.fatten(m_skin_thickness);

211  auto& node = m_nodes.at(node_index);
212  node.aabb = aabb;
213  node.aabb.update_area();
214 
215  insert_leaf(node_index);
216  return true;
217  }
218  else
219  {
220  return false;
221  }

259  inline void rebuild()
260  {
261  std::vector<index_type> node_indices(m_node_count);
262  int count {0};
263 
264  for (auto index = 0; index < m_node_capacity; ++index)
265  {
266  if (m_nodes.at(index).height < 0)
267  {
268  // Free node.
269  continue;
270  }
271 
272  if (m_nodes.at(index).is_leaf())
273  {
274  m_nodes.at(index).parent = std::nullopt;
275  node_indices.at(count) = index;
276  ++count;
277  }
278  else
279  {
280  free_node(index);
281  }
282  }
283 
284  while (count > 1)
285  {
286  auto min_cost = std::numeric_limits<double>::max();
287  int iMin {-1};
288  int jMin {-1};
289 
290  for (auto i = 0; i < count; ++i)
291  {
292  const auto fst_aabb = m_nodes.at(node_indices.at(i)).aabb;
293 
294  for (auto j = (i + 1); j < count; ++j)
295  {
296  const auto snd_aabb = m_nodes.at(node_indices.at(j)).aabb;
297  const auto cost = AABB::merge(fst_aabb, snd_aabb).area();
298 
299  if (cost < min_cost)
300  {
301  iMin = i;
302  jMin = j;
303  min_cost = cost;
304  }
305  }
306  }
307 
308  const auto index1 = node_indices.at(iMin);
309  const auto index2 = node_indices.at(jMin);
310 
311  const auto parent_index = allocate_node();
312  auto& parent_node = m_nodes.at(parent_index);
313 
314  auto& index1Node = m_nodes.at(index1);
315  auto& index2Node = m_nodes.at(index2);
316 
317  parent_node.left = index1;
318  parent_node.right = index2;
319  parent_node.height = 1 + std::max(index1Node.height, index2Node.height);
320  parent_node.aabb = AABB::merge(index1Node.aabb, index2Node.aabb);
321  parent_node.parent = std::nullopt;
322 
323  index1Node.parent = parent_index;
324  index2Node.parent = parent_index;
325 
326  node_indices.at(jMin) = node_indices.at(count - 1);
327  node_indices.at(iMin) = parent_index;
328  --count;
329  }
330 
331  m_root = node_indices.at(0);
332  }

369  template<size_type buffer_size = 256, typename OutputIterator>
370  inline void query(const key_type& key, OutputIterator iterator) const
371  {
372  if (const auto it = m_index_map.find(key); it != m_index_map.end())
373  {
374  const auto& source_node = m_nodes.at(it->second);
375 
376  std::array<std::byte, sizeof(maybe_index) * buffer_size> buffer;
377  std::pmr::monotonic_buffer_resource resource {buffer.data(), sizeof buffer};
378 
379  pmr_stack<maybe_index> stack {&resource};
380  stack.push(m_root);
381  while (!stack.empty())
382  {
383  const auto node_index = stack.top();
384  stack.pop();
385 
386  if (!node_index)
387  {
388  continue;
389  }
390 
391  const auto& node = m_nodes.at(*node_index);
392 
393  // Test for overlap between the AABBs
394  if (source_node.aabb.overlaps(node.aabb, m_touch_is_overlap))
395  {
396  if (node.is_leaf() && node.id)
397  {
398  if (node.id != key)
399  {
400  // Can't interact with itself
401  *iterator = *node.id;
402  ++iterator;
403  }
404  }
405  else
406  {
407  stack.push(node.left);
408  stack.push(node.right);
409  }
410  }
411  }
412  }

441  [[nodiscard]] inline const double compute_surface_area_ratio() const
442  {
443  if (m_root == std::nullopt)
444  {
445  return 0;
446  }
447 
448  const auto root_area = m_nodes.at(*m_root).aabb.compute_area();
449  double total_area = 0.0;

452  {

559  inline void grow_pool()
560  {
561  // The free list is empty. Rebuild a bigger pool.
562  m_node_capacity *= 2;
563  resize_to_match_node_capacity(m_node_count);
564

566  m_next_free_index = static_cast<index_type>(m_node_count);

572  [[nodiscard]] inline const index_type allocate_node()
573  {
574  if (m_next_free_index == std::nullopt)
575  {
576  grow_pool();
577  }
578 
579  // Peel a node off the free list.
580  const auto node_index = m_next_free_index.value();
581  auto& node = m_nodes.at(node_index);
582 
583  m_next_free_index = node.next;
584  node.parent = std::nullopt;
585  node.left = std::nullopt;
586  node.right = std::nullopt;
587  node.height = 0;

609  [[nodiscard]] static inline const double left_cost(const AABB& leaf_aabb, const node_type& left_node, const double minimum_cost)
610  {
611  if (left_node.is_leaf())
612  {
613  return AABB::merge(leaf_aabb, left_node.aabb).area() + minimum_cost;
614  }
615  else
616  {
617  const auto old_area = left_node.aabb.area();
618  const auto new_area = AABB::merge(leaf_aabb, left_node.aabb).area();
619  return (new_area - old_area) + minimum_cost;
620  }
621  }
622 
626  [[nodiscard]] static inline const double right_cost(const AABB& leaf_aabb, const node_type& right_node, const double minimum_cost)
627  {
628  if (right_node.is_leaf())
629  {
630  const auto aabb = AABB::merge(leaf_aabb, right_node.aabb);
631  return aabb.area() + minimum_cost;
632  }
633  else
634  {
635  const auto aabb = AABB::merge(leaf_aabb, right_node.aabb);
636  const auto old_area = right_node.aabb.area();
637  const auto new_area = aabb.area();
638  return (new_area - old_area) + minimum_cost;
639  }
640  }

645  [[nodiscard]] inline const index_type find_best_sibling(const AABB& leaf_aabb) const
646  {
647  auto index = m_root.value();
648 
649  while (!m_nodes.at(index).is_leaf())
650  {
651  const auto& node = m_nodes.at(index);
652  const auto left = node.left.value();
653  const auto right = node.right.value();
654 
655  const auto surface_area = node.aabb.area();
656  const auto combined_surface_area =
657  AABB::merge(node.aabb, leaf_aabb).area();
658 
659  // Cost of creating a new parent for this node and the new leaf.
660  const auto cost = 2.0 * combined_surface_area;
661 
662  // Minimum cost of pushing the leaf further down the tree.
663  const auto minimum_cost = 2.0 * (combined_surface_area - surface_area);
664 
665  const auto cost_left = left_cost(leaf_aabb, m_nodes.at(left), minimum_cost);
666  const auto cost_right =
667  right_cost(leaf_aabb, m_nodes.at(right), minimum_cost);
668 
669  // Descend according to the minimum cost.
670  if ((cost < cost_left) && (cost < cost_right))
671  {
672  break;
673  }
674 
675  // Descend.
676  if (cost_left < cost_right)
677  {
678  index = left;
679  }
680  else
681  {
682  index = right;
683  }
684  }
685 
686  return index;
687  }

692  [[nodiscard]] inline const index_type balance(const index_type node_index)
693  {
694  if (m_nodes.at(node_index).is_leaf() || (m_nodes.at(node_index).height < 2))
695  {
696  return node_index;
697  }
698 
699  const auto left_index = m_nodes.at(node_index).left.value();
700  const auto right_index = m_nodes.at(node_index).right.value();
701 
702  const auto current_balance = m_nodes.at(right_index).height - m_nodes.at(left_index).height;
703 
704  // Rotate right branch up.
705  if (current_balance > 1)
706  {
707  rotate_right(node_index, left_index, right_index);
708  return right_index;
709  }
710 
711  // Rotate left branch up.
712  if (current_balance < -1)
713  {
714  rotate_left(node_index, left_index, right_index);
715  return left_index;
716  }
717 
718  return node_index;
719  }

724  inline void fix_tree_upwards(maybe_index index)
725  {
726  while (index != std::nullopt)
727  {
728  index = balance(*index);
729

731

733  const auto right = node.right.value();
734 
735  const auto& left_node = m_nodes.at(left);
736  const auto& right_node = m_nodes.at(right);
737 
738  node.height = 1 + std::max(left_node.height, right_node.height);
739  node.aabb = AABB::merge(left_node.aabb, right_node.aabb);
740 
741  index = node.parent;
742  }

748  inline void insert_leaf(const index_type leaf_index)
749  {
750  if (m_root == std::nullopt)
751  {
752  m_root = leaf_index;
753  m_nodes.at(*m_root).parent = std::nullopt;
754  return;
755  }
756 
757  // Find the best sibling for the node.
758  const auto leaf_aabb = m_nodes.at(leaf_index).aabb; // copy current AABB
759  const auto sibling_index = find_best_sibling(leaf_aabb);
760 
761  // Create a new parent.
762  const auto old_parent_index = m_nodes.at(sibling_index).parent;
763  const auto new_parent_index = allocate_node();
764 
765  auto& new_parent = m_nodes.at(new_parent_index);
766  new_parent.parent = old_parent_index;
767  new_parent.aabb = AABB::merge(leaf_aabb, m_nodes.at(sibling_index).aabb);
768 
769  new_parent.height = m_nodes.at(sibling_index).height + 1;
770 
771  if (old_parent_index != std::nullopt)
772  {
773  // The sibling was not the root.
774  auto& old_parent = m_nodes.at(*old_parent_index);
775  if (old_parent.left == sibling_index)
776  {
777  old_parent.left = new_parent_index;
778  }
779  else
780  {
781  old_parent.right = new_parent_index;
782  }
783  }
784  else
785  {
786  // The sibling was the root.
787  m_root = new_parent_index;
788  }
789 
790  new_parent.left = sibling_index;
791  new_parent.right = leaf_index;
792 
793  m_nodes.at(sibling_index).parent = new_parent_index;
794  m_nodes.at(leaf_index).parent = new_parent_index;
795 
796  // Walk back up the tree fixing heights and AABBs.
797  fix_tree_upwards(m_nodes.at(leaf_index).parent);
798  }

826  inline void remove_leaf(const index_type leaf_index)
827  {
828  if (leaf_index == m_root)
829  {
830  m_root = std::nullopt;
831  return;
832  }
833 
834  const auto parent_index = m_nodes.at(leaf_index).parent;
835  const auto grand_parent_index = m_nodes.at(parent_index.value()).parent;
836 
837  const auto sibling_index =
838  (m_nodes.at(parent_index.value()).left == leaf_index)
839  ? m_nodes.at(parent_index.value()).right
840  : m_nodes.at(parent_index.value()).left;
841 
842  // Destroy the parent and connect the sibling to the grandparent.
843  if (grand_parent_index != std::nullopt)
844  {
845  if (m_nodes.at(*grand_parent_index).left == parent_index)
846  {
847  m_nodes.at(*grand_parent_index).left = sibling_index;
848  }
849  else
850  {
851  m_nodes.at(*grand_parent_index).right = sibling_index;
852  }
853 
854  m_nodes.at(sibling_index.value()).parent = grand_parent_index;
855  free_node(parent_index.value());
856 
857  // Adjust ancestor bounds.
858  adjust_ancestor_bounds(grand_parent_index);
859  }
860  else
861  {
862  m_root = sibling_index;
863  m_nodes.at(sibling_index.value()).parent = std::nullopt;
864  free_node(parent_index.value());
865  }
866  }

871  inline void rotate_right(const index_type node_index, const index_type left_index, const index_type right_index)
872  {
873  auto& node = m_nodes.at(node_index);
874  auto& right_node = m_nodes.at(right_index);
875 
876  const auto right_left = right_node.left.value();
877  const auto right_right = right_node.right.value();
878 
879  // Swap node and its right-hand child.
880  right_node.left = node_index;
881  right_node.parent = node.parent;
882  node.parent = right_index;
883 
884  // The node's old parent should now point to its right-hand child.
885  if (right_node.parent != std::nullopt)
886  {
887  auto& right_parent = m_nodes.at(*right_node.parent);
888  if (right_parent.left == node_index)
889  {
890  right_parent.left = right_index;
891  }
892  else
893  {
894  right_parent.right = right_index;
895  }
896  }
897  else
898  {
