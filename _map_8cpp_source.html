405  void Map::parse_tilesets()
406  {
407  for (auto& tileset : m_tilesets)
408  {
409  const auto tileset_name = std::filesystem::path(tileset.get_image()).stem().string();
410  const auto& tileset_region = SL_HANDLE.atlas()->get_region(tileset_name);
411 
412  for (const auto& tile : tileset.get_tiles())
413  {
414  const int rows = tileset.get_tile_count() / tileset.get_columns();
415  const int column = tile.get_id() % tileset.get_columns();
416  const int row = tile.get_id() / tileset.get_columns();
417 
418  auto u = (column != 0) ? (column * tileset.get_tile_width()) : (0 * tileset.get_tile_width());
419  auto v = (row < rows - 1) ? (row * tileset.get_tile_height()) : ((rows - 1) * tileset.get_tile_height());
420 
421  u += ((column * tileset.get_spacing()) + tileset.get_margin());
422  v += ((row * tileset.get_spacing()) + tileset.get_margin());
423 
424  u += tileset_region.m_x;
425  v += tileset_region.m_y;
426 
427  const auto name = tileset_name + std::to_string(tile.get_id());
428  graphics::fRect rect;
429  rect.m_x = u;
430  rect.m_y = v;
431  rect.m_width = tileset.get_tile_width();
432  rect.m_height = tileset.get_tile_height();
433 
434  SL_HANDLE.atlas()->add_custom_region(name, rect);
435  }
436  }

439  const int Map::parse_layers(const nlohmann::json& json, int level)
440  {
441  if (json.count("layers") > 0)
442  {
443  const auto& layer_array = json.at("layers");
444  for (const auto& layer : layer_array)
445  {
446  if (layer.count("type") > 0)
447  {
448  const auto& type = layer.at("type");
449  if (type == "tilelayer")
450  {
451  m_tile_layers.emplace_back(layer, level);
452  level++;
453  }
454  else if (type == "objectgroup")
455  {
456  m_object_layers.emplace_back(layer, level);
457  level++;
458  }
459  else if (type == "imagelayer")
460  {
461  m_image_layers.emplace_back(layer, level);
462  level++;
463  }
464  else if (type == "group")
465  {
466  level = parse_layers(layer, level);
467  }
468  }
469  }
470  }
471 
472  return level;
473  }

501  void Map::create_object_layer(const ObjectLayer& layer, core::World& world)
502  {
503  if (layer.get_draworder() == "topdown")
504  {
505  const auto& objects = layer.get_objects();
506  for (const auto& object : objects)
507  {
508  const auto type = object.get_type_enum();
509 
510  if (type == Object::Type::TEXT)
511  {
512  GALAXY_LOG(GALAXY_WARNING, "Text objects are not supported.");
513  }
514  else
515  {
516  const auto entity = world.create();
517  auto* transform = world.create_component<components::Transform>(entity);
518  auto* shaderid = world.create_component<components::ShaderID>(entity);
519  auto* primitive2d = world.create_component<components::Primitive2D>(entity);
520  auto* renderable = world.create_component<components::Renderable>(entity);
521  renderable->m_z_level = layer.get_z_level();
522 
523  switch (type)
524  {
525  case Object::Type::ELLIPSE:
526  {
527  components::Primitive2D::PrimitiveData data;
528  data.m_colour = layer.get_colour();
529  data.m_fragments = std::make_optional(40);
530  data.m_radii = std::make_optional<glm::vec2>(object.get_width() / 2.0, object.get_height() / 2.0);
531 
532  primitive2d->create<graphics::Primitives::ELLIPSE>(data);
533 
534  transform->set_pos(object.get_x(), object.get_y());
535  transform->rotate(object.get_rotation());
536 
537  renderable->m_type = graphics::Renderables::LINE_LOOP;
538  shaderid->m_shader_id = "line";
539  }
540  break;
541 
542  case Object::Type::POINT:
543  {
544  components::Primitive2D::PrimitiveData data;
545  data.m_colour = layer.get_colour();
546  data.m_pointsize = 3;
547  primitive2d->create<graphics::Primitives::POINT>(data);
548 
549  transform->set_pos(object.get_x(), object.get_y());
550  transform->rotate(object.get_rotation());
551 
552  renderable->m_type = graphics::Renderables::POINT;
553  shaderid->m_shader_id = "point";
554  }
555  break;
556 
557  case Object::Type::POLYGON:
558  {
559  components::Primitive2D::PrimitiveData data;
560  std::vector<glm::vec2> points;
561  for (const auto& point : object.get_points())
562  {
563  points.emplace_back(point.get_x(), point.get_y());
564  }
565 
566  data.m_points = std::make_optional(points);
567  data.m_colour = layer.get_colour();
568  primitive2d->create<graphics::Primitives::POLYGON>(data);
569 
570  transform->set_pos(object.get_x(), object.get_y());
571  transform->rotate(object.get_rotation());
572 
573  renderable->m_type = graphics::Renderables::LINE_LOOP;
574  shaderid->m_shader_id = "line";
575  }
576  break;
577 
578  case Object::Type::POLYLINE:

580  components::Primitive2D::PrimitiveData data;
581  std::vector<glm::vec2> points;
582  for (const auto& point : object.get_points())
583  {
584  points.emplace_back(point.get_x(), point.get_y());
585  }
586 
587  data.m_points = std::make_optional(points);
588  data.m_colour = layer.get_colour();
589  primitive2d->create<graphics::Primitives::POLYLINE>(data);
590 
591  transform->set_pos(object.get_x(), object.get_y());
592  transform->rotate(object.get_rotation());
593

595  shaderid->m_shader_id = "line";
596  }
597  break;
598 
599  case Object::Type::RECT:
600  {
601  components::Primitive2D::PrimitiveData data;
602  std::vector<glm::vec2> points;
603  points.emplace_back(0.0f, 0.0f);
604  points.emplace_back(0.0f + object.get_width(), 0.0f);
605  points.emplace_back(0.0f + object.get_width(), 0.0f + object.get_height());
606  points.emplace_back(0.0f, 0.0f + object.get_height());
607 
608  data.m_points = std::make_optional(points);
609  data.m_colour = layer.get_colour();
610  primitive2d->create<graphics::Primitives::POLYGON>(data);
611 
612  transform->set_pos(object.get_x(), object.get_y());
613  transform->rotate(object.get_rotation());
614 
615  renderable->m_type = graphics::Renderables::LINE_LOOP;
616  shaderid->m_shader_id = "line";
617  }
618  break;
619  }
620 
621  if (!object.get_name().empty())
622  {
623  auto* tag = world.create_component<components::Tag>(entity);
624  tag->m_tag = object.get_name();
625  }
626 
627  if (object.is_visible())
628  {
629  world.enable(entity);
630  }
631 
632  m_object_entities.push_back(entity);
633  }
634  }
635  }

642  void Map::create_tile_layer(const TileLayer& layer, core::World& world)
643  {

652  m_data[level]->m_batch->set_texture(SL_HANDLE.atlas()->get_atlas());
653  }
654 
655  const auto& data = layer.get_data();
656  for (int i = 0; i < m_height; i++)

664  const auto gid = unset_tile_flags(flagged_gid);

668  const auto tile_name = tileset_name + std::to_string(tile->get_id());
669 
670  if (!tile->get_animations().empty())
671  {
672  const auto entity = world.create();
673  auto* renderable = world.create_component<components::Renderable>(entity);
674  auto* batch = world.create_component<components::BatchSprite>(entity);

676  auto* transform = world.create_component<components::Transform>(entity);
677 
678  renderable->m_z_level = layer.get_z_level();
679  renderable->m_type = graphics::Renderables::BATCHED;
680 
681  batch->create(tile_name, layer.get_opacity());
682 
683  std::vector<graphics::Frame> frames;
684  frames.reserve(tile->get_animations().size());
685 
686  for (const auto& tile_anim : tile->get_animations())
687  {

690 
691  const auto tile_anim_name = tileset_name + std::to_string(tile_anim.get_tile_id());
692  frame.set_region(tile_anim_name);
693 
694  frames.emplace_back(frame);
695  }
696 
697  animated->add_animation("TileAnimation", "TileAnimation", true, 1.0, frames);
698  animated->set_animation("TileAnimation");
699  animated->play();
700 
701  transform->set_pos(layer.get_offset_x() + (j * tileset->get_tile_width()), layer.get_offset_y() + (i * tileset->get_tile_height()));
702 
703  world.enable(entity);
704  }
705  else
706  {
707  auto& batch_data = m_data[level]->m_batch_data.emplace_back(std::make_unique<BatchData>());
708  batch_data->first = std::make_unique<components::BatchSprite>();
709  batch_data->second = std::make_unique<components::Transform>();
710  batch_data->first->create(tile_name, layer.get_opacity());
711  batch_data->second->set_pos(layer.get_offset_x() + (j * tileset->get_tile_width()), layer.get_offset_y() + (i * tileset->get_tile_height()));
712 
713  auto& end = m_data[level]->m_batch_data.back();
714  m_data[level]->m_batch->add(end->first.get(), end->second.get(), level);
715  }
716  }
