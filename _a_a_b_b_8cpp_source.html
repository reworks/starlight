65  void AABB::fatten(const std::optional<double>& factor) noexcept
66  {
67  if (!factor)
68  {
69  return;
70  }
71 
72  const auto size = m_max - m_min;
73  const auto dx = *factor * size.x;
74  const auto dy = *factor * size.y;
75 
76  m_min.x -= dx;
77  m_min.y -= dy;
78 
79  m_max.x += dx;
80  m_max.y += dy;
81 
82  m_area = compute_area();
83  }

127  const double AABB::compute_area() const noexcept
128  {
129  // Sum of "area" of all the sides.
130  auto sum = 0.0;
131 
132  // hold one dimension constant and multiply by all the other ones.
133  for (auto d1 = 0; d1 < 2; ++d1)
134  {
135  auto product = 1.0; // "Area" of current side.
136 
137  for (auto d2 = 0; d2 < 2; ++d2)
138  {
139  if (d1 != d2)
140  {
141  const auto dx = m_max[d2] - m_min[d2];
142  product *= dx;
143  }
144  }
145 
146  sum += product;
147  }
148 
149  return 2.0 * sum;
150  }

172  AABB AABB::merge(const AABB& fst, const AABB& snd) noexcept
173  {

176 
177  for (auto i = 0; i < 2; ++i)
